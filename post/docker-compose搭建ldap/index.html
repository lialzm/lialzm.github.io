version: '3.3'
services:
  openldap:
    image: osixia/openldap
    container_name: openldap
    environment:
      - TZ=Asia/Shanghai
      - LDAP_ORGANISATION="admin"
      - LDAP_DOMAIN="admin.com"
      - LDAP_ADMIN_PASSWORD="123456"
    ports:
      - 389:389
      - 636:636
    command: [--copy-service,  --loglevel, debug]
    networks:
      test:
    volumes:
      - ./ldap/database:/var/lib/ldap
      - ./ldap/config:/etc/ldap/slapd.d
  php:
    image: osixia/phpldapadmin
    container_name: phpopenldap
    environment:
      - TZ=Asia/Shanghai
      - PHPLDAPADMIN_HTTPS="false"
      - PHPLDAPADMIN_LDAP_HOSTS=openldap
    ports:
      - 10004:80
    depends_on:
      - openldap
    networks:
      test:
    links:
      - openldap
networks:
  test:
