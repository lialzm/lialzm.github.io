job "hello1" {
  datacenters = ["dc1"]  #定义数据中心

  group "hello2" {      #组名字
    task "hello3" {      #一般使用服务名字表示task名字
      driver = "docker"     #使用docker驱动

      config {
        image = "hashicorp/http-echo"    #服务镜像名字
        args = [                                        #容器运行时的命令参数
          "-listen", ":5678",
          "-text", "hello world",
        ]
      }

      resources {                                #配置服务的资源
        network {
          mbits = 10                            #限制10MB带宽
          port "http" {
            static = "5678"                    #使用静态端口
          }
        }
      }
    }
  }
}
